    # Get orders

    # Request

    Url: https://api.vektor-adv.com/api/orders
    Method: POST
    Type: JSON
    Headers:

    • Authorization: Bearer [YOUR_ACCESS_TOKEN]
    • Content-Type: application/json
    # BODY:
    • orderId (uuid, required) - generate client side
    • customer (object, required)
      • customer.email (string, required)
      • customer.name (string)
      • customer.phone (string)
    • certificates (array, required)
      • certificates[0].id (uuid, required) - get from Get certificates
      • certificates[0].quantity (int,required)
    # Example:
    {
   "orderId": "B5AA69DF-61A5-4D97-B1FB-4F6772DFBBEB",
   "customer": {
       "email": "test@test.test",
       "phone": "79251153242",         
       "name": "John Doe",
   },
   "certificates": [
       {
           "id": "A25714EA-F9E6-4D53-94FC-6F9D0587BB9D",
           "quantity": 1
       },
       {
           "id": "53AF1480-1D33-41CA-A08E-0BDC4BD318D6",
           "quantity": 1
       }
   ]
}

    # Response

    Type: json

    Status: 200 (OK)

    # Errors

    Status: 400 (Bad request)

    {
 "message": "Invalid request",
 "status": 400,
 "errors": {
   "Customer.Email": [
     "The Email field is required."
   ]
 }
}
