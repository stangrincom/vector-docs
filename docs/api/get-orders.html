    # Get access token

    # Request

    Url: https://api.vektor-adv.com/connect/token
    Method: POST
    Type: form-data (!)not json

    # BODY:
    • client_id (string, required) - your client id
    • client_secret (string, required) - your client secret
    • grant_type (string, required) = client_credentials
    # Example:
    client_id: [your client id]
client_secret: [your client secret]
grant_type: client_credentials

    # Response

    Type: json

    # BODY:
    • access_token (string)
    • expires_in (int) - time live token in second
    • token_type (string)
    • scope (string)

    Status: 200 (OK)

    # Example:
    {
    "access_token": "eyJhbGciOiJSUzI1NiIsImtpZCI6IlJ2eURPZEF...",
    "expires_in": 3600,
    "token_type": "Bearer",
    "scope": "external-api"
}

    # Errors:

    Status: 400 (Bad request)

    {
    "error": "invalid_request"
}

    # Create order

    # Request

    Url: https://api.vektor-adv.com/api/orders
    Method: POST
    Type: JSON
    Headers:

    • Authorization: Bearer [YOUR_ACCESS_TOKEN]
    • Content-Type: application/json
    # BODY:
    • orderId (uuid, required) - generate client side
    • customer (object, required)
      • customer.email (string, required)
      • customer.name (string)
      • customer.phone (string)
    • certificates (array, required)
      • certificates[0].id (uuid, required) - get from Get certificates
      • certificates[0].quantity (int,required)
    # Example:
    {
   "orderId": "B5AA69DF-61A5-4D97-B1FB-4F6772DFBBEB",
   "customer": {
       "email": "test@test.test",
       "phone": "79251153242",         
       "name": "John Doe",
   },
   "certificates": [
       {
           "id": "A25714EA-F9E6-4D53-94FC-6F9D0587BB9D",
           "quantity": 1
       },
       {
           "id": "53AF1480-1D33-41CA-A08E-0BDC4BD318D6",
           "quantity": 1
       }
   ]
}

    # Response

    Type: json

    Status: 200 (OK)

    # Errors

    Status: 400 (Bad request)

    {
 "message": "Invalid request",
 "status": 400,
 "errors": {
   "Customer.Email": [
     "The Email field is required."
   ]
 }
}
