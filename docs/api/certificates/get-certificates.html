    # Get certificates

    # Request

    Url: https://api.vektor-adv.com/api/certificates
    Method: GET
    Type: JSON
    Headers:

    • Authorization: Bearer [YOUR_ACCESS_TOKEN]

    # Response

    # BODY:
    • id (uuid)
    • vendor (string)
    • unit (string) - nominal unit (ruble, month)
    • value (decimal) - nominal value
    • price (decimal)
